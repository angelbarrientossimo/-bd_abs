```python
from pyspark.sql.functions import split,col,lit,regexp_replace,cast

df = (
    df
    .withColumn("amount_aux1",split(col("Amount")," ")[0])
    .withColumn("amount_aux2",split(col("Amount")," ")[1])
    .withColumn("amount_aux2",regexp_replace(col("amount_aux2"),"Lac",lit(100000)))
    .withColumn("amount_aux2",regexp_replace(col("amount_aux2"),"Cr",lit(10000000)))
    .withColumn("Amount",col("amount_aux1") * col("amount_aux2"))
    .withColumn("Amount_USD",(col("Amount") * 0.012).cast("float"))
    .withColumn("area_aux",split(col("Carpet Area")," ")[1])
    .withColumn("Carpet Area",split(col("Carpet Area")," ")[0])
    .withColumn("area_aux",regexp_replace(col("area_aux"),"sqft",lit(0.0929)))
    .withColumn("Carpet Area",col("Carpet Area") * col("area_aux"))
)
```
