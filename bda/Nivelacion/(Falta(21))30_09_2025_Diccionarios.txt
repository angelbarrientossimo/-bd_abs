3.- Contador de frecuencias de palabras
Escribe un programa que tome una frase y use un diccionario para contar la frecuencia de cada palabra.

def contar_frecuencia(frase):
    palabras = frase.lower().split()
    frecuencia = {}
    for palabra in palabras:
        palabra = palabra.strip(".,;:!?¡¿()[]{}\"'")
        if palabra in frecuencia:
            frecuencia[palabra] += 1
        else:
            frecuencia[palabra] = 1
    
    return frecuencia


frase = "¡Hola mundo! hola ¿cómo estás mundo?"
resultado = contar_frecuencia(frase)
print(resultado)

21.- Invertir mayúsculas y minúsculas en claves
Dado un diccionario con claves tipo string, crea otro donde las claves estén con las letras invertidas (mayúsculas ↔ minúsculas).


diccionario = {
    "nombre": "Carlos",
    "edAd": 28,
    "ciUdad": "Madrid",
    "prOfesión": "Ingeniero",
    "hobbies": ["leer", "viajar", "correr"],
    "activo": True,
    "salario": 3500.50,
    "fecha_registro": "2023-10-05"
}
dict_aux = {}
for keys, values in diccionario.items():
    cadena_aux = ""
    for caracter in keys:
        if caracter == caracter.lower():
            caracter = caracter.upper()
            cadena_aux += caracter
        else:
            caracter = caracter.lower()
            cadena_aux += caracter
    dict_aux[cadena_aux] = values
print(dict_aux)
