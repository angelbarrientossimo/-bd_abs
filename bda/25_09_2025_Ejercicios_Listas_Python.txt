33. Suma acumulada

Ejemplo: [1,2,3,4] → [1,3,6,10].



def suma_acumulada(lista):
    acumulada = []
    suma = 0
    for numero in lista:
        suma += numero
        acumulada.append(suma)
    return acumulada

entrada = [1, 2, 3, 4]
resultado = suma_acumulada(entrada)
print(resultado)
